# install

vagrant up

mysqladmin create nairbl

sudo nano /etc/nginx/nginx.conf

in file set path to folder /var/www/default/public

cp .env.example .env

php artisan jwt:secret

set APP_KEY in .env file

sudo service nginx restart

php artisan migrate

php vendor/bin/phpunit
